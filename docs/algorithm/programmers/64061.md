```js
function solution(board, moves) {
  const stack = [];
  let answer = 0;

  moves.forEach((order) => {
    const target = pick(board, order - 1);
    if (target) {
      if (stack[stack.length - 1] === target) {
        stack.pop();
        answer += 2;
      } else {
        stack.push(target);
      }
    }
  });
  return answer;
}

function pick(board, order) {
  for (let i = 0; i < board.length; i++) {
    if (board[i][order] !== 0) {
      target = board[i][order];
      board[i][order] = 0;
      return target;
    }
  }
}
```

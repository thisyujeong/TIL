const [t, ...rest] = require('fs')
  .readFileSync('./input/11021.txt')
  .toString()
  .split('\n');
let result = '';
for (let i = 0; i < Number(t); i++) {
  const [a, b] = rest[i].split(' ').map(Number);
  result += `Case #${i + 1}: ${a + b} \n`;
}

  const visited = new Array(size + 1).fill(0);
  for (let i = 1; i <= size; i++) {
    if (visited[i] === 0) {
      dfs(i);
      answer++;
    }
  }

  /* DFS */

    if (visited[perm[v]] === 0) {
      dfs(perm[v]);
    }
  }
  console.log(answer);
});

## Solution

```js
const fs = require('fs');
const filePath = process.platform === 'linux' ? '/dev/stdin' : '../input.txt';
let [N, A, B] = fs.readFileSync(filePath).toString().trim().split('\n');

let sum = 0;
N = Number(N);
A = A.split(' ')
  .map(Number)
  .sort((a, b) => a - b);
B = B.split(' ')
  .map(Number)
  .sort((a, b) => b - a);

for (let i = 0; i < N; i++) {
  sum += A[i] * B[i];
}

console.log(sum);
```
